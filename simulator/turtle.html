# Спираль import turtle t = turtle.Turtle() t.speed(0) for i in range(100): t.forward(i) t.right(91)
# Цветок import turtle t = turtle.Turtle() t.speed(0) for i in range(36): for j in range(4): t.forward(100) t.right(90) t.right(10)
# Разноцветные круги import turtle t = turtle.Turtle() t.speed(0) colors = ['red', 'orange', 'yellow', 'green', 'blue', 'purple'] for i in range(36): t.color(colors[i % 6]) t.circle(100) t.left(10)
